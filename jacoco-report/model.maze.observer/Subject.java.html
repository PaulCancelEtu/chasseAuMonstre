package model.maze.observer;

import java.util.HashSet;
import java.util.Set;

/**
 * Cette classe représente un objet observable, soit un sujet d'observations.
 * 
 * <p>Un {@code Subject} peut être observé par un ou plusieurs {@code IObserver}.</p>
 * 
 * <p>Lorsqu'un {@code Subject} change, il peut appelé sa méthode {@code notifyObservers}
 * pour informer tous ses observateurs du changement en appelant leur méthode {@code update}.</p>
 */

public abstract class Subject {
	protected Set<IObserver> observers = new HashSet<>();
	
	/**
	 * Invoque la méthode {@code update} sans paramètre de tous les observateurs attachés.
	 * 
	 */
	protected void notifyObservers() {
		for(IObserver obs : this.observers)
		{
			obs.update(this);
		}
	}

	/**
	 * Invoque la méthode {@code update} de tous les observateurs attachés
	 * en passant un paramètre.
	 * 
	 * @param data n'importe quel objet
	 */
	protected void notifyObservers(Object data) 
	{
		for(IObserver obs : this.observers)
		{
			obs.update(this,data);
		}
	}

	/**
	 * Ajoute un observateur à l'ensemble d'observateurs de cet objet.
	 * 
	 * @param observer l'observateur à ajouter
	 */
	public void attach(IObserver observer) {
		this.observers.add(observer);
	}

	/**
	 * Retire un observateur à l'ensemble des observateurs de cet objet,
	 * s'il y est présent.
	 * 
	 * @param observer l'observateur à retirer de la liste
	 */
	public void detach(IObserver observer) {
		this.observers.remove(observer);
	}
}
