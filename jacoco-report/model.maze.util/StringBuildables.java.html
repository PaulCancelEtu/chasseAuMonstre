package model.maze.util;

/**
 * Cette classe contient la méthode {@code stringBuildableToString}, utilisable
 * pour toute classe souhaitant utiliser le principe du {@code StringBuilder} dans
 * sa méthode {@code toString}.
 */

public class StringBuildables {
	private StringBuildables() {}
	
	/**
	 * Renvoie l'objet passé en paramètre représenté sous la forme d'une chaîne de caractères, 
	 * selon sa méthode {@code fillStringBuilder}.
	 * 
	 * <p>Le but de cette méthode est d'éviter la concaténation à tout-va de {@code String}
	 * en passant par un {@code StringBuilder}.</p>
	 * 
	 * @param object l'objet dont on souhaite avoir la représentation sous la forme d'une chaîne de caractères
	 * @return l'objet passé en paramètre représenté sous la forme d'une chaîne de caractères
	 */
	public static String stringBuildableToString(IStringBuildable object) {
		StringBuilder sb = new StringBuilder();
		sb.append(object.getClass().getSimpleName());
		sb.append('[');

		object.fillStringBuilder(sb);

		sb.append(']');

		return sb.toString();
	}
}
