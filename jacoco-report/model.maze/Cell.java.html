package model.maze;

import fr.univlille.iutinfo.cam.player.perception.ICellEvent;
import fr.univlille.iutinfo.cam.player.perception.ICellEvent.CellInfo;

/**
 * Une cellule est une case dans le tableau de notre jeu.
 * 
 * <p>Elle a connaissance du dernier évènement lui concernant, et
 * dispose d'un état fixe.</p>
 * 
 * <p>Par défaut, le dernier évènement est {@code null}.</p>
 */

public class Cell extends Coordinate {
	private final CellInfo STATE;
	private ICellEvent lastEvent;

	protected Cell(int row, int col, CellInfo state) {
		super(row, col);
		this.STATE = state;
		this.lastEvent = null;
	}

	/**
	 * @return le tour du dernier évènement de la cellule, {@code -1} 
	 *         si aucun évènement n'a jamais eu lieu sur cette cellule
	 */
	public int getTurn() {
		if (this.lastEvent == null)
			return CellEvent.INIT_TURN;
		return this.lastEvent.getTurn();
	}

	/**
	 * @return l'état selon le dernier évènement de la cellule, {@code null} 
	 *         si aucun évènement n'a jamais eu lieu sur cette cellule
	 */
	public CellInfo getUpdatedState() {
		if (this.lastEvent == null)
			return null;
		return this.lastEvent.getState();
	}

	/**
	 * Retourne l'état immutable, c'est-à-dire sans considération du dernier
	 * évènement de la cellule.
	 * 
	 * @return l'état immutable de notre cellule
	 */
	public CellInfo getState() {
		return this.STATE;
	}

	/**
	 * Indique si la cellule est <i>traversable</i>, c'est-à-dire si
	 * le monstre a le droit de se déplacer sur cette cellule.
	 * 
	 * <p>Une cellule est traversable si et seulement si son état immutable
	 * est autre que {@code CellInfo.WALL}.</p>
	 * 
	 * @return si la cellule est traversable ou non
	 */
	public boolean isTravelable() {
		return this.STATE != CellInfo.WALL;
	}

	/**
	 * Indique si la cellule est une sortie.
	 * 
	 * <p>Une cellule est une si et seulement si son état immutable
	 * est {@code CellInfo.EXIT}.</p>
	 * 
	 * @return si la cellule est une sortie ou non
	 */
	public boolean isExit() {
		return this.STATE == CellInfo.EXIT;
	}
	
	/**
	 * Écrase le dernier évènement concernant la cellule.
	 * 
	 * <p>Aucune vérification concernant la correspondance des 
	 * coordonnées de l'évènement à celles de la cellule n'est effectuée.</p>
	 * 
	 * @param event le nouvel évènement concernant la cellule
	 */
	void setLastEvent(ICellEvent event) {
		this.lastEvent = event;
	}

	@Override
	public void fillStringBuilder(StringBuilder sb) {
		super.fillStringBuilder(sb);

		sb.append(';');
		sb.append(this.STATE);

		if (this.lastEvent != null) {
			sb.append(';');
			sb.append(this.lastEvent);
		}
	}

	@Override
	public boolean equals(Object obj) {
		if (obj == null)
			return false;
		if (obj == this)
			return true;
		if (!(obj instanceof Cell))
			return false;

		Cell other = (Cell) obj;

		if (this.lastEvent == null) {
			if (other.lastEvent != null) return false;
		} else if (!this.lastEvent.equals(other.lastEvent)) {
			return false;
		}

		return super.equals(other) && this.STATE == other.STATE;
	}
}
