package model.maze;

import fr.univlille.iutinfo.cam.player.perception.ICellEvent;
import fr.univlille.iutinfo.cam.player.perception.ICoordinate;
import model.maze.util.StringBuildable;
/**
 * Classe implémentant l'interface {@code ICellEvent}.
 */
public class CellEvent extends StringBuildable implements ICellEvent {
	static final int INIT_TURN = -1;

	private int turn;
	private final CellInfo STATE;
	private final ICoordinate COORD;

	/**
	 * Construit un évènement de cellule.
	 * 
	 * <p>Le tour de l'évènement est à -1 par défaut et n'est pas
	 * paramètrable à l'instanciation pour toute manipulation indésirable.</p>
	 * 
	 * @param state l'état de la cellule
	 * @param coord les coordonnées de la cellule
	 */
	public CellEvent(CellInfo state, ICoordinate coord) {
		this.turn = INIT_TURN;
		this.COORD = coord;
		this.STATE = state;
	}

	@Override
	public CellInfo getState() {
		return this.STATE;
	}

	@Override
	public int getTurn() {
		return this.turn;
	}

	@Override
	public ICoordinate getCoord() {
		return this.COORD;
	}

	void setTurn(int turn) {
		this.turn = turn;
	}

	@Override
	public void fillStringBuilder(StringBuilder sb) {
		sb.append(this.STATE);
		sb.append(';');
		sb.append(this.turn);
		sb.append(";[");
		sb.append(this.COORD.getRow());
		sb.append(", ");
		sb.append(this.COORD.getCol());
		sb.append(']');
	}

	@Override
	public boolean equals(Object obj) {
		if (obj == null)
			return false;
		if (obj == this)
			return true;
		if (!(obj instanceof CellEvent))
			return false;

		CellEvent other = (CellEvent) obj;

		return this.STATE == other.STATE
				&& this.turn == other.turn
				&& this.COORD.getRow() == other.COORD.getRow()
				&& this.COORD.getCol() == other.COORD.getCol();
	}
}
