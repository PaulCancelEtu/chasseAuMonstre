package model.maze;

import java.io.File;
import java.io.FileNotFoundException;
import java.util.Scanner;

import fr.univlille.iutinfo.cam.player.perception.ICellEvent.CellInfo;

/**
 * Cette énurmération est temporaire.
 * 
 * <p>Un {@code Level} encapsule la grille d'un niveau sous la forme d'un
 * tableau bidimensionnel de {@code CellInfo}.</p>
 */

public enum Level {
    LEVEL_1(LevelUtils.importLevel1()),
    LEVEL_2(LevelUtils.importLevel2()),
    LEVEL_3(LevelUtils.importLevel3());

    private CellInfo[][] grid;

    private Level(CellInfo[][] grid) {
        this.grid = grid;
    }

    public int getNbRows(){
        return this.grid.length;
    }

    public int getNbCols(){
        if(this.grid == null) return 0;
        return this.grid[0].length;
    }

    // public CellInfo[][] getGrid() {
    //     return grid;
    // }

    public CellInfo getCellInfo(int row, int col){
        return this.grid[row][col];
    }



    private static class LevelUtils{
        private static final File LEVEL_1_FILE = new File("res/level1.txt");
        private static final File LEVEL_2_FILE = new File("res/level2.txt");
        private static final File LEVEL_3_FILE = new File("res/level3.txt");

        private static final CellInfo[][]importLevel1(){
            return importGrid(10, 10, LEVEL_1_FILE);
        }
        private static final CellInfo[][]importLevel2(){
            return importGrid(10, 10, LEVEL_2_FILE);
        }
        private static final CellInfo[][]importLevel3(){
            return importGrid(10, 10, LEVEL_3_FILE);
        }


        private static CellInfo[][] importGrid(int height, int width, File file){
            CellInfo[][] res = new CellInfo[height][width];
            try(Scanner sc = new Scanner(file)){
                for(int i = 0; i < height; i++){
                    res[i] = importLine(width, sc.nextLine());
                }
            }catch(FileNotFoundException e){
                e.printStackTrace();
                return res;
            }
            return res;
        }


        private static CellInfo[] importLine(int size, String line){
            CellInfo[] res = new CellInfo[size];
            for(int i = 0; i<size; i++){
                res[i] = importCell(line.charAt(i));
            }

            return res;
        }

        private static CellInfo importCell(char character){
            switch(character){
                case '_': return CellInfo.EMPTY;
                case '*': return CellInfo.WALL;
                case 'm': return CellInfo.MONSTER;
                case 's': return CellInfo.EXIT;
                default: return null;
            }
        }
    }
}
