package model.maze;

import fr.univlille.iutinfo.cam.player.perception.ICoordinate;
import model.maze.util.StringBuildable;

/**
 * Simple classe implémentant l'interface {@code ICoordinate}.
 * 
 * <p>Une {@code Coordinate} est immutable : la ligne et la colonne
 * sont définies à l'instanciation.</p>
 */

public class Coordinate extends StringBuildable implements ICoordinate {
	private final int ROW;
	private final int COL;

	public Coordinate(int row, int col) {
		ROW = row;
		COL = col;
	}

	@Override
	public int getRow() {
		return ROW;
	}

	@Override
	public int getCol() {
		return COL;
	}

	@Override
	public void fillStringBuilder(StringBuilder sb) {
		sb.append(this.getRow());
		sb.append(", ");
		sb.append(this.getCol());
	}

	@Override
	public boolean equals(Object obj) {
		if (obj == null)
			return false;
		if (obj == this)
			return true;
		if (!(obj instanceof Coordinate))
			return false;

		Coordinate other = (Coordinate) obj;

		return other.ROW == this.ROW && other.COL == this.COL;
	}
}
