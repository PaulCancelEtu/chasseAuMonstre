package model.maze.roles;

import fr.univlille.iutinfo.cam.player.IStrategy;
import fr.univlille.iutinfo.cam.player.perception.ICellEvent;
import fr.univlille.iutinfo.cam.player.perception.ICoordinate;
import model.maze.Game;
import model.maze.observer.IObserver;
import model.maze.observer.Subject;
import model.maze.util.StringBuildableSubject;

/**
 * Classe représentant un acteur dans le jeu.
 */

public abstract class Role extends StringBuildableSubject implements IObserver {
	protected IStrategy strategy;

	/**
	 * Joue en exécutant la méthode {@code play} de la stratégie attribuée.
	 * 
	 * @throws NoStrategyFoundException si aucune stratégie n'a été attribuée
	 */
	public void play() {
		if (this.hasStrategy()) {
			ICoordinate coord = this.strategy.play();
			this.play(coord.getRow(), coord.getCol());
		} else {
			throw new NoStrategyFoundException();
		}
	}

	/**
	 * Créé un CellEvent selon les coordonnés données et notifie les observateurs attachés.
	 * 
	 * @param row la ligne de la case à jouer
	 * @param col la colonne de la case à jouer
	 */
	public abstract void play(int row, int col);

	/**
	 * Appelle la méthode {@code update} de la stratégie attribuée, si elle est attribuée.
	 * 
	 * <p>Cette méthode est invoquée par {@code Game} lorsqu'un tour a été validé.</p>
	 */
	protected void updateStrategy(ICellEvent event) {
		if (this.hasStrategy()) {
			this.strategy.update(event);
		}
	}

	/**
	 * Indique si une stratégie est attribuée au rôle.
	 * 
	 * @return si une stratégie est attribuée au rôle
	 */
	public boolean hasStrategy() {
		return this.strategy != null;
	}

	@Override
	public void update(Subject subj) {}

	@Override
	public void update(Subject subj, Object data) {
		if (subj instanceof Game && data instanceof ICellEvent) {
			this.updateStrategy((ICellEvent) data);
		}
	}
}
