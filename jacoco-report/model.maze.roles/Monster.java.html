package model.maze.roles;

import fr.univlille.iutinfo.cam.player.monster.IMonsterStrategy;
import fr.univlille.iutinfo.cam.player.perception.ICellEvent;
import fr.univlille.iutinfo.cam.player.perception.ICoordinate;
import fr.univlille.iutinfo.cam.player.perception.ICellEvent.CellInfo;
import model.maze.Cell;
import model.maze.CellEvent;
import model.maze.Coordinate;
import model.maze.Game;
import model.maze.observer.Subject;

/**
 * Classe représentant le monstre.
 */

public class Monster extends Role {
	private IMonsterStrategy strategy;
	private ICoordinate coord;

	public Monster(IMonsterStrategy strategy, ICoordinate coord) {
		this.strategy = strategy;
		this.coord = coord;
	}

	public Monster(ICoordinate coord) {
		this(null, coord);
	}

	/**
	 * Indique la stratégie du monstre
	 * @return la stratégie du monstre
	 */
	public IMonsterStrategy getStrategy() {
		return strategy;
	}

	/**
	 * Indique les coordonnées sur lesquelles se trouve le monstre
	 * @return les coordonnées sur lesquelles se trouve le monstre
	 */
	public ICoordinate getCoord() {
		return this.coord;
	}

	private void setCoord(ICoordinate coord) {
		this.coord = coord;
	}

	@Override
	public void play(int row, int col) {
		CellEvent event = new CellEvent(CellInfo.MONSTER, new Coordinate(row, col));
		this.notifyObservers(event);
	}

	@Override
	public void update(Subject subj, Object data) {
		super.update(subj, data);

		if (subj instanceof Game && data instanceof ICellEvent) {
			Game game = (Game) subj;

			if (!game.getMonster().equals(this)) return;
			
			ICellEvent event = (ICellEvent) data;

			this.setCoord(event.getCoord());
		}
	}

	@Override
	public void fillStringBuilder(StringBuilder sb) {
			sb.append(this.strategy);
			sb.append(';');
			sb.append(this.coord.toString());
	}

	@Override
	public boolean equals(Object obj) {
		if (obj == null)
			return false;
		if (obj == this)
			return true;
		if (!(obj instanceof Monster))
			return false;

		Monster other = (Monster) obj;

		if (this.strategy == null) {
			if (other.strategy != null) return false;
		} else if (!this.strategy.equals(other.strategy)) {
			return false;
		}

		if (this.coord == null) {
			if (other.coord != null) return false;
		} else if (!this.coord.equals(other.coord)) {
			return false;
		}

		return true;
	}
	
	/**
	 * Indique si les coordonnées données sont à la portée du monstre.
	 * 
	 * <p>Retourne vrai si les coordonnées sont adjacentes et ne sont pas 
	 * en diagonnale par rapport aux coordonnées du monstre.</p>
	 * 
	 * @param target les coordonnées en question
	 * @return si les coordonnées données sont à la portée du monstre
	 */
	public boolean isWithinRange(ICoordinate target) {
		int mRow = this.coord.getRow();
		int mCol = this.coord.getCol();

		int tRow = target.getRow();
		int tCol = target.getCol();

		return Math.abs((mRow - tRow) + (mCol - tCol)) == 1;
	}

	/**
	 * Indique si le monstre peut se déplacer dans la cellule donnée.
	 * 
	 * <p>Retourne vrai si la cellule est traversable, qu'elle est à la portée
	 * du monstre et que le monstre ne s'y trouve pas.</p>
	 * 
	 * @param cell la cellule
	 * @return si le monstre peut se déplacer dans la cellule donnée
	 */
	public boolean canPlay(Cell cell) {
		return cell.isTravelable() 
			&& this.isWithinRange(cell) 
			&& !coord.equals(cell);
	}
}
