package model.maze.roles;

import fr.univlille.iutinfo.cam.player.hunter.IHunterStrategy;
import fr.univlille.iutinfo.cam.player.perception.ICellEvent.CellInfo;
import model.maze.CellEvent;
import model.maze.Coordinate;

/**
 * Classe représentant le chasseur.
 */

public class Hunter extends Role {
	private IHunterStrategy strategy;

	public Hunter(IHunterStrategy strategy) {
		this.strategy = strategy;
	}

	public Hunter() {
		this(null);
	}

	/**
	 * Indique la stratégie du chasseur
	 * @return la stratégie du chasseur
	 */
	public IHunterStrategy getStrategy() {
		return strategy;
	}

	@Override
	public void play(int row, int col) {
		CellEvent event = new CellEvent(CellInfo.HUNTER, new Coordinate(row, col));
		this.notifyObservers(event);
	}

	@Override
	public void fillStringBuilder(StringBuilder sb) {
		sb.append(this.strategy);
	}
	
	@Override
	public boolean equals(Object obj) {
		if (obj == null)
			return false;
		if (obj == this)
			return true;
		if (!(obj instanceof Hunter))
			return false;

		Hunter other = (Hunter) obj;

		if (this.strategy == null) {
			if (other.strategy != null) return false;
		} else if (!this.strategy.equals(other.strategy)) {
			return false;
		}

		return true;
	}
}
